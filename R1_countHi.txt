//Recursion-1 > countHi
public int countHi(String str) {
  int len = str.length();
  if(len <= 1) return 0;
  else{
    len--;
    if(str.toLowerCase().charAt(len) == 'i' && str.toLowerCase().charAt(len-1) == 'h'){
      return 1 + countHi(str.substring(0, len));
    }
    else return countHi(str.substring(0, len));
  }
}
