//Recursion-1 > array6
public boolean array6(int[] nums, int index) {
  int len = nums.length;
  if (len == 0) return false;
  else {
    if(index < len){
      if(nums[index] == 6){
        return true;
      }
      else {
        index++;
        return array6(nums, index);
      }
    }
    else {
      return false;
    }
  }
}
